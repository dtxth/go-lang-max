```go
func (uc *MigrateFromExcelUseCase) processExcelStreaming(
    ctx context.Context, 
    jobID int, 
    filePath string, 
    processed *int, 
    failed *int
) error {
    // Open file
    f, err := excelize.OpenFile(filePath)
    if err != nil {
        return err
    }
    defer f.Close()

    // Get streaming iterator
    rows, err := f.Rows(sheetName)
    if err != nil {
        return err
    }
    defer rows.Close()

    // Process rows one by one
    for rows.Next() {
        row, err := rows.Columns()
        if err != nil {
            continue
        }

        // Parse and validate
        excelRow := parseRow(row)
        
        // Process immediately
        if err := uc.processRow(ctx, jobID, &excelRow); err != nil {
            (*failed)++
        } else {
            (*processed)++
        }

        // Update progress every 100 rows
        if (*processed + *failed) % 100 == 0 {
            uc.jobRepo.UpdateProgress(ctx, jobID, *processed, *failed)
        }
    }

    return rows.Error()
}
```
